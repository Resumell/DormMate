一、C语言结构体

/* 上行：传感器 + 环境数据 + 事件上下文 */
typedef struct {
    int     light;              // 光敏传感器 ADC 原始值 0-4095，值越大越暗
    int     human;              // 人体红外检测 0=无人 1=有人
    float   temperature;        // 温度 ℃（DHT11）
    float   humidity;           // 湿度 %（DHT11）
    float   current;            // 电流 单位安培（预留，当前未接硬件，固定 0.5）
    int     relay;              // 台灯状态 0=关 1=开
    int     ac_relay;           // 空调灯状态 0=关 1=开
    int     servo;              // 舵机角度 0-180，-1=未启用（预留）
    char    current_time[6];    // 当前时间 "HH:MM"
    char    history[7][6];      // 最近7天回寝时间（用于AI学习）
    int     predict_query;      // 1=本次上报为预处理预测查询
    char    event[32];          // 事件类型："human_return"/"human_first"/""
    char    date[16];           // 日期 "YYYY-MM-DD"
    char    first_seen_time[6]; // 首次检测到人的时间 "HH:MM"
    int     weekday;            // 星期几 0=周日
    char    source[16];         // 触发来源："infrared"/"voice"/"manual"
} SensorData_t;

/* 下行：后端决策指令 */
typedef struct {
    int     relay;              // 台灯: -1=不操作, 0=关, 1=开
    int     ac_relay;           // 空调灯: -1=不操作, 0=关, 1=开
    int     servo;              // 舵机角度，-1=不操作（预留）
    int     buzzer;             // 蜂鸣器 0=静音 1=响（预留，当前未接硬件）
    char    tts[64];            // TTS 播报文本，空字符串=不播报
    char    oled_line1[32];     // OLED 第一行（仅英文/数字/符号）
    char    oled_line2[32];     // OLED 第二行
    int     preheat;            // 预处理标记 1=触发空调预处理
    int     target_temp;        // 目标温度（预处理用）
    char    predict_return_time[6]; // 预测回寝时间 "HH:MM"
    char    reason[64];         // 决策原因（日志用）
    int     voice_notify;       // 语音播报编号：
                                //   0=不播报
                                //   1=询问是否开灯
                                //   4=询问是否开空调
                                //   9=欢迎回家
                                //   10=预处理通知（已开启环境预处理）
                                //   11=设置已修改
    int     manual;             // 1=前端手动控制触发（彩虹灯等特效用）
} ActionData_t;

二、上行JSON（ESP32 → 后端）

{
    "device_id": "dormmate_01",
    "sensors": {
        "light": 850,
        "human": 1,
        "temperature": 26.5,
        "humidity": 60.0,
        "current": 0.5
    },
    "status": {
        "relay": 0,
        "ac_relay": 0,
        "servo": -1
    },
    "context": {
        "event": "human_return",
        "date": "2026-07-15",
        "first_seen_time": "21:30",
        "weekday": 2,
        "source": "infrared",
        "query_type": "predict",
        "current_time": "21:30",
        "history": ["21:30", "21:15", "21:45", "", "", "", ""]
    }
}

字段说明：
- context.event：""=普通环境上报；"human_return"=红外回寝事件；"human_first"=首次回寝入库
- context.query_type：""=普通上报；"predict"=预处理预测查询
- context.source："infrared"=红外触发；"voice"=语音触发

三、下行JSON（后端 → ESP32）

{
    "actions": {
        "relay": -1,
        "ac_relay": -1,
        "servo": -1,
        "buzzer": 0,
        "tts": "当前光线太暗，是否要为您开灯",
        "oled_line1": "AI:Light?",
        "oled_line2": "",
        "preheat": 0,
        "target_temp": 24,
        "predict_return_time": "21:30",
        "reason": "本地fallback：用户设定时间21:30",
        "voice_notify": 1,
        "manual": 0
    }
}

字段说明：
- relay/ac_relay：-1 表示不改变当前状态，0/1 强制开关
- voice_notify：ESP32 收到后通过 UART 发送给 SU-03T 语音模块
- manual：1 时 ESP32 会执行彩虹灯等前端手动特效

四、后端API接口

1  ESP32 数据上报（内部用）
   POST http://电脑IP:5000/api/report
   Body：上行JSON
   返回：下行JSON

2  ESP32 单次运行模式（内部用）
   POST http://电脑IP:5000/api/run_once
   Body：上行JSON
   返回：{"actions": ..., "mode": "run_once"}

3  获取后端状态
   GET  http://电脑IP:5000/api/status
   返回：{"msg": "...", "ai_online": true/false, "model": "..."}

4  获取预测与偏好状态（手机端用）
   GET  http://电脑IP:5000/api/predict/status?device_id=dormmate_01
   返回：{
       "today": "2026-07-15",
       "predict_time": "21:30",
       "preheat_triggered": true/false,
       "actual_return_time": "21:30",
       "preference": {"pref_temp": 24, "preheat_enable": 1, "custom_return_time": "21:30"},
       "history": [...]
   }

5  设置用户偏好（手机端用）
   POST http://电脑IP:5000/api/preference/return
   Body JSON：
   {
       "device_id": "dormmate_01",
       "pref_temp": 24,
       "preheat_enable": 1,
       "custom_return_time": "21:30"
   }
   返回：{"status": "ok", "actions": {"voice_notify": 11, "tts": "已为您修改设置"}}

6  手动触发预处理（手机端用）
   POST http://电脑IP:5000/api/preheat/trigger
   Body JSON：{"device_id": "dormmate_01"}
   返回：{"status": "ok", "msg": "已触发，设备下次上报时执行预处理"}

7  获取回寝历史（手机端用）
   GET  http://电脑IP:5000/api/history/return?device_id=dormmate_01&days=30
   返回：{"history": [{"date": "...", "first_seen_time": "...", "temp_at_arrival": ...}]}

8  获取当前传感器（手机端用）
   GET  http://电脑IP:5000/api/current
   返回：{"sensors": {"light": ..., "human": ..., "temperature": ..., "humidity": ...}}

9  前端控制设备（手机端用）
   POST http://电脑IP:5000/api/control
   Body JSON：{"device_id": "dormmate_01", "relay": 1, "servo": 90, "buzzer": 0}
   返回：{"status": "ok", "pending": {"relay": 1}}

10 获取服务器时间（ESP32对时用）
    GET  http://电脑IP:5000/api/time
    返回：{"year": ..., "month": ..., "day": ..., "hour": ..., "minute": ..., "second": ..., "epoch": ...}
