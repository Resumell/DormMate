一、C语言结构体，驱动函数命名不限，能跑通就行

typedef struct {
    int     light;              // 光敏传感器 0-4095  值越小越暗
    int     human;              // 人体雷达 0=无人 1=有人
    float   temp;               // 温度 单位摄氏度
    float   humi;               // 湿度 单位百分比
    float   current;            // 电流 单位安培  用于大功率检测
    int     relay;              // 继电器状态 0=断开 1=吸合
    int     servo;              // 舵机角度 0-180  -1表示未启用
    int     buzzer;             // 蜂鸣器 0=静音 1=响
    char    ir_code[16];        // 红外编码字符串  空字符串=不发红外
    char    oled_line1[32];     // OLED第一行显示内容
    char    oled_line2[32];     // OLED第二行显示内容
} SensorData_t;

二、上行JSON（ESP32上报给后端）

{
    "device_id": "dorm_001",            // 设备唯一标识，字符串
    "timestamp": 1720156800,            // Unix时间戳，秒级
    
    "sensors": {
        "light": 850,                   // 光敏 0-4095
        "human": 1,                     // 人体 0/1
        "temp": 26.5,                   // 温度
        "humi": 60.0,                   // 湿度
        "current": 0.8                  // 电流
    },
    
    "state": {
        "relay": 1,                     // 继电器当前状态
        "servo": 90,                    // 舵机当前角度
        "buzzer": 0,                    // 蜂鸣器是否正在响
        "ir_code": "",                  // 最后一次发送的红外码
        "oled_line1": "Mode:Auto",      // OLED第一行
        "oled_line2": "L:850 T:26.5"    // OLED第二行
    }
}

三、下行JSON（后端下发给ESP32）

{
    "device_id": "dorm_001",            // 目标设备，防止串号
    "timestamp": 1720156805,
    
    "actions": {
        "relay": 1,                     // 0=断开 1=吸合  -1=保持不动
        "servo": 90,                    // 0-180角度      -1=保持不动
        "buzzer": 1,                    // 0=静音 1=响     -1=保持不动
        "ir_code": "FFA25D",            // 红外编码字符串  空字符串=不发
        "tts": "光线不足，已为您开灯",   // 喇叭要说的文字  空字符串=不播报
        "oled_line1": "AI:开灯",        // OLED第一行更新
        "oled_line2": "Target:24C"      // OLED第二行更新
    },
    
    "source": "auto",                   // 决策来源：auto=自动  ai=AI决策  manual=手动
    "need_confirm": false                // true=需要用户语音确认后再执行  false=直接执行
}

四、后端API接口（给手机端调用）

1  获取实时状态
   GET  http://电脑IP:5000/api/status
   返回：上面的上行JSON格式（最新一条）
   
2  获取历史数据
   GET  http://电脑IP:5000/api/history?hours=24
   参数：hours=查询最近几小时，默认24
   返回：JSON数组，包含多条上行记录
   
3  获取用户偏好
   GET  http://电脑IP:5000/api/preferences
   返回：偏好规则列表
   
4  修改用户偏好
   POST http://电脑IP:5000/api/preference
   Body JSON：
   {
       "scene": "study",               // 场景名：study sleep return
       "trigger": "light < 200",       // 触发条件，字符串
       "action": "turn_on_light",      // 执行动作
       "auto_execute": true            // true=自动执行  false=仅记录
   }
   返回：{"status": "ok", "id": 3}
   
5  删除偏好
   DELETE http://电脑IP:5000/api/preference?id=3
   返回：{"status": "ok"}

6  ESP32上报（内部用，C不需要调）
   POST http://电脑IP:5000/api/report
   Body：上行JSON

7  AI决策请求（内部用，C不需要调）
   POST http://电脑IP:5000/api/ai_decision
   Body：上行JSON
   返回：下行JSON
